Lets build a C++ triangle pattern!

```
#include<iostream>using namespace std;int main(){ int i, x = 5, y , z = 1; while (x > 0){for ( i = 0; i < x; i ++){ cout <<"  "; } for (y = 0 ; y < z; y++) {cout << " +"; }cout << endl; x--;z+=2;}system ("PAUSE"); return 0;}
```
